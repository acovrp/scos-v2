function wowInlineBadge(curr, prev, metric) {
  if (curr === null || prev === null || Math.abs(prev) < 0.0001) return '';
  var pct = (curr - prev) / Math.abs(prev) * 100;
  var lowerBetter = (metric==='acos'||metric==='tacos'||metric==='adspend');
  var good = lowerBetter ? pct < 0 : pct > 0;
  var cls = Math.abs(pct) < 1 ? 'neu' : (good ? 'pos' : 'neg');
  var arrow = pct > 0.05 ? '▲' : pct < -0.05 ? '▼' : '→';
  return '<span class="delta '+cls+'" style="display:block;font-size:9px;margin-top:1px">'+arrow+Math.abs(pct).toFixed(1)+'%</span>';
}

function getConsolidatedVal(prod, metric, col) {
  var channels = ['amz','flk','qcom'];
  var vals = channels.map(function(ch) {
    return prod.channels[ch] ? getVal(prod.channels[ch], metric, col) : null;
  }).filter(function(v){ return v !== null; });
  if (!vals.length) return null;
  if (isRateMetric(metric)) {
    var weights = channels.map(function(ch) {
      return prod.channels[ch] ? getVal(prod.channels[ch], 'units', col) : null;
    }).filter(function(v){return v!==null;});
    if (weights.length === vals.length && weights.some(function(w){return w>0;})) {
      var wSum=0,wTot=0; vals.forEach(function(v,i){var w=weights[i]||1;wSum+=v*w;wTot+=w;});
      return wTot>0 ? wSum/wTot : vals.reduce(function(a,b){return a+b},0)/vals.length;
    }
    return vals.reduce(function(a,b){return a+b},0)/vals.length;
  }
  return vals.reduce(function(a,b){return a+b}, 0);
}

function getDisplayVal(prod, ch, metric, col) {
  if (S.view === 'channel') return getVal(prod.channels[ch], metric, col);
  return getConsolidatedVal(prod, metric, col);
}

function fmtV(v, metric) {
  if (v === null || v === undefined) return '<span style="color:#d1d5db">—</span>';
  if (metric === 'revenue' || metric === 'adspend') {
    if (Math.abs(v) >= 10000000) return '₹'+(v/10000000).toFixed(2)+'Cr';
    if (Math.abs(v) >= 100000)   return '₹'+(v/100000).toFixed(1)+'L';
    if (Math.abs(v) >= 1000)     return '₹'+(v/1000).toFixed(0)+'K';
    return '₹'+Math.round(v);
  }
  if (metric === 'units') return Math.round(v).toLocaleString('en-IN');
  if (isRateMetric(metric)) return v.toFixed(metric==='ctr'||metric==='cvr'||metric==='adcvr'?2:1)+'%';
  return v.toFixed(1);
}

function trendBadge(vals, metric) {
  var recent = vals.slice(-4).filter(function(v){return v!==null});
  var older  = vals.slice(0,4).filter(function(v){return v!==null});
  if (!recent.length || !older.length) return '—';
  var rA = recent.reduce(function(a,b){return a+b},0)/recent.length;
  var oA = older.reduce(function(a,b){return a+b},0)/older.length;
  var pct = ((rA-oA)/Math.abs(oA)*100).toFixed(1);
  var better = (metric === 'acos' || metric === 'tacos') ? parseFloat(pct)<0 : parseFloat(pct)>0;
  var color = better ? '#28a745' : Math.abs(parseFloat(pct)) < 2 ? '#6c757d' : '#dc3545';
  var arrow = better ? '↑' : Math.abs(parseFloat(pct)) < 2 ? '→' : '↓';
  return '<span style="color:'+color+';font-weight:700">'+arrow+' '+Math.abs(pct)+'%</span>';
}

function getCell(lineOrCat, wk) {
    var keys = Object.keys(WEEKLY_CUBE).filter(function(k){
      var p = k.split('||');
      if (groupBy === 'cat') {
        var prod = PRODUCTS.find(function(x){ return x.line === p[0]; });
        return prod && prod.cat === lineOrCat && p[1] === wk;
      }
      return p[0] === lineOrCat && p[1] === wk;
    });
    if (!keys.length) return null;
    var c = {revenue:0, units:0, sessions:0, pageViews:0, adSpend:0, adSales:0, impressions:0, adClicks:0, orders:0, adUnits:0};
    keys.forEach(function(k){
      var d = WEEKLY_CUBE[k];
      for (var f in c) if (d[f]) c[f] += d[f];
    });
    c.acos = c.adSales > 0 ? c.adSpend / c.adSales * 100 : 0;
    c.roas = c.adSpend > 0 ? c.adSales / c.adSpend : 0;
    c.ctr  = c.impressions > 0 ? c.adClicks / c.impressions * 100 : 0;
    c.cvr  = c.adClicks > 0 ? c.orders / c.adClicks * 100 : 0;
    c.tacos = c.revenue > 0 ? c.adSpend / c.revenue * 100 : 0;
    c.organicPct = c.revenue > 0 ? (c.revenue - c.adSales) / c.revenue * 100 : 0;
    c.cvrBiz = c.sessions > 0 ? c.units / c.sessions * 100 : 0;
    return c;
  }

function isoWeekFromDate(dtStr) {
  var parts = dtStr.match(/(\d{4})[-\/](\d{1,2})[-\/](\d{1,2})/);
  if (!parts) return null;
  var dt = new Date(parseInt(parts[1]), parseInt(parts[2])-1, parseInt(parts[3]));
  if (isNaN(dt)) return null;
  dt.setHours(0,0,0,0);
  dt.setDate(dt.getDate() + 3 - (dt.getDay() + 6) % 7);
  var jan4 = new Date(dt.getFullYear(), 0, 4);
  var wk = 1 + Math.round(((dt - jan4) / 86400000 - 3 + (jan4.getDay() + 6) % 7) / 7);
  return dt.getFullYear() + '-W' + (wk < 10 ? '0' : '') + wk;
}

function processFkAdsReport(rows) {
  var hs=rows[0].map(function(h){return h.trim().toLowerCase().replace(/[^a-z0-9_ ]+/g,'');});
  function ci(){for(var i=0;i<arguments.length;i++)for(var j=0;j<hs.length;j++)if(hs[j].indexOf(arguments[i])>=0)return j;return -1;}
  var iSpend=ci('spend','cost','budget spent','amount spent'), iTitle=ci('product name','title','listing','campaign name');
  if (iSpend<0) { console.warn('FKADS: no spend column'); return; }
  var total=0, byProd={};
  for (var i=1;i<rows.length;i++) {
    var r=rows[i], sp=parseFloat((r[iSpend]||'0').replace(/[^\d.]/g,''))||0;
    total+=sp;
    if (iTitle>=0) { var pid=matchFkProduct(r[iTitle]||''); if (pid) byProd[pid]=(byProd[pid]||0)+sp; }
  }
  window.UPLOADED_TOTALS=window.UPLOADED_TOTALS||{}; window.UPLOADED_TOTALS.fkads=Math.round(total); window.UPLOADED_TOTALS.fkads_by_prod=byProd;
  // Write per-product adspend + TACoS to PRODUCTS.flk for table display
  for (var pid in byProd) {
    var p=PRODUCTS.find(function(x){return x.id===pid;}); if (!p||!p.channels.flk) continue;
    var wk=Math.round(byProd[pid]/4);
    if (!p.channels.flk.adspend) p.channels.flk.adspend={};
    p.channels.flk.adspend.recent=[wk,wk,wk,wk];
    if (!p.channels.flk.tacos) p.channels.flk.tacos={};
    var rev=(p.channels.flk.revenue.recent||[]).reduce(function(s,v){return s+(v||0);},0);
    var tc=rev>0?Math.round(byProd[pid]/rev*1000)/10:0;
    p.channels.flk.tacos.recent=[tc,tc,tc,tc];
  }
  console.log('FKADS spend: ₹'+Math.round(total)); renderFlkKpi(); renderTables();
}

function getRow(wk) {
    var base;
    if (type === 'SP') {
      // productWise = SP-only API data; use total (all products aggregated)
      base = pw[wk] ? pwToRow(pw[wk].total) : null;
    } else if (type === 'SD' || type === 'SB') {
      var wkd = tv[wk]; if (!wkd) return null;
      base = wkd.types ? wkd.types[type] : null;
    } else {
      // 'all': prefer typeView (has SP+SB+SD), fall back to productWise total for newer weeks
      if (tv[wk]) { base = tv[wk].total; }
      else if (pw[wk]) { base = pwToRow(pw[wk].total); }
    }
    if (!base) return null;
    var r = Object.assign({}, base);
    if (r.acos != null && r.acos < 5) r.acos = r.acos * 100;
    return r;
  }

  return wowBuildTable(weeks, cols, getRow, 'aza_');
}
